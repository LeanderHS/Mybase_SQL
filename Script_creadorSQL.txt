DROP DATABASE IF EXISTS ArteyVida;
CREATE DATABASE ArteyVida;
USE ArteyVida;

CREATE TABLE Entrada (
    	CodEnt INT AUTO_INCREMENT PRIMARY KEY,
    	Estado ENUM('Disponible','Vendida','Cancelada') NOT NULL DEFAULT 'Disponible',
   	PrecioEnt DECIMAL(8,2) NOT NULL,
	CodSec SMALLINT NOT NULL,
	FOREIGN KEY (CodSec) REFERENCES Seccion (CodSec)
		ON DELETE CASCADE
		ON UPDATE CASCADE
);

CREATE TABLE Asistente (
   	Dni VARCHAR(9) PRIMARY KEY,
    	NombreC VARCHAR(50) NOT NULL,
    	Email VARCHAR(50) NOT NULL UNIQUE,
    	CONSTRAINT chk_email_format CHECK (Email REGEXP '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$')
);


CREATE TABLE Compran (
    	CodCompra INT AUTO_INCREMENT PRIMARY KEY,
   	 CodEnt INT NOT NULL,
   	 Dni VARCHAR(9) NOT NULL,
    	Valoran TINYINT CHECK (Valoran BETWEEN 1 AND 5),
    	FecComp DATE NOT NULL,
    	UNIQUE (CodEnt, Dni),
    	FOREIGN KEY (CodEnt) REFERENCES Entrada(CodEnt)
        		ON DELETE CASCADE
        		ON UPDATE CASCADE,
    	FOREIGN KEY (Dni) REFERENCES Asistente(Dni)
        		ON DELETE RESTRICT
        		ON UPDATE CASCADE
);

CREATE TABLE Telefono (
Dni VARCHAR(9) NOT NULL,
    	CodPais VARCHAR(5) NOT NULL,
    	Numero VARCHAR(15) NOT NULL,
    	PRIMARY KEY (Dni, Numero),
    	FOREIGN KEY (Dni) REFERENCES Asistente(Dni)
        		ON DELETE CASCADE
        		ON UPDATE CASCADE,
    	CONSTRAINT chk_numero_formato CHECK (Numero REGEXP '^[0-9]{6,15}$')
);

CREATE TABLE Seccion (
    	CodSec SMALLINT PRIMARY KEY,
    	Tipo ENUM('VIP', 'Estándar') NOT NULL,
    	AforoSec INT NOT NULL CHECK (AforoSec > 0),
    	MgSec DECIMAL(3,2) NOT NULL CHECK (MgSec BETWEEN 0 AND 1),
    	CodUb SMALLINT NOT NULL,
   	 FOREIGN KEY (CodUb) REFERENCES Ubicacion(CodUb)
        		ON DELETE CASCADE
       		 ON UPDATE CASCADE
);

CREATE TABLE Ubicacion (
    	CodUb SMALLINT AUTO_INCREMENT PRIMARY KEY,
    	Nombre VARCHAR(100) NOT NULL,
   	 Direccion VARCHAR(100),
    	Ciudad VARCHAR(30) NOT NULL,
    	TipoUb ENUM('Ciudad', 'Pueblo') NOT NULL,
     	PrecioAlq DECIMAL(8,2) NOT NULL,
    	Caract VARCHAR(200)
);

CREATE TABLE Actividad (
    	CodAct SMALLINT AUTO_INCREMENT PRIMARY KEY,
   	Nombre VARCHAR(100) NOT NULL,
    	Tipo ENUM('Música','Teatro','Danza','Cine','Pintura','Escultura','Literatura',
'Fotografía') NOT NULL
);

CREATE TABLE Artista (
    	CodArt SMALLINT AUTO_INCREMENT PRIMARY KEY,
Nombre VARCHAR(50) NOT NULL,
   	 Apellido1 VARCHAR(50) NOT NULL,
    	Apellido2 VARCHAR(50),
       	Biblio TEXT
);

CREATE TABLE Participan (
    	CodAct SMALLINT NOT NULL,
    	CodArt SMALLINT NOT NULL,
   	 Cache DECIMAL(8,2) NOT NULL CHECK (Cache >= 0),
   	 PRIMARY KEY (CodAct, CodArt),
    	FOREIGN KEY (CodAct) REFERENCES Actividad(CodAct)
        		ON DELETE RESTRICT
        		ON UPDATE CASCADE,
   	FOREIGN KEY (CodArt) REFERENCES Artista(CodArt)
       		ON DELETE RESTRICT
        		ON UPDATE CASCADE
);

CREATE TABLE Evento (
    	CodEven SMALLINT AUTO_INCREMENT PRIMARY KEY,
    	Nombre VARCHAR(100) NOT NULL,
    	Fecha DATE NOT NULL CHECK (Fecha >= CURDATE()),
    	Hora TIME NOT NULL,
    	CodAct SMALLINT NOT NULL,
    	FOREIGN KEY (CodAct) REFERENCES Actividad(CodAct)
       		ON DELETE RESTRICT
        		ON UPDATE CASCADE,
   );

CREATE TABLE Reserva (
    	CodEven SMALLINT NOT NULL,
   	CodSec SMALLINT NOT NULL,
    	FecRes DATE NOT NULL CHECK (FecRes <= CURDATE()),
    	PRIMARY KEY (CodSec, CodEven),
    	FOREIGN KEY (CodEven) REFERENCES Evento(CodEven)
        		ON DELETE CASCADE
        		ON UPDATE CASCADE,
    	FOREIGN KEY (CodSec) REFERENCES Seccion(CodSec)
        		ON DELETE CASCADE
        		ON UPDATE CASCADE
) ;
